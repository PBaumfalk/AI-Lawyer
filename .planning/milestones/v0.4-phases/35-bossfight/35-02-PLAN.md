**Initial fetch (useEffect):**
```typescript
useEffect(() => {
  let cancelled = false;
  async function fetchBossState() {
    try {
      const res = await fetch("/api/gamification/bossfight");
      if (cancelled) return;
      if (!res.ok) { setLoaded(true); return; }
      const data = await res.json();
      if (data.active) {
        setBoss(data.boss);
        setLeaderboard(data.leaderboard ?? []);
        setRecentDamage(data.recentDamage ?? []);
        setTeaser(null);
      } else {
        setTeaser(data.teaser);
        setBoss(null);
      }
      setLoaded(true);
    } catch {
      setLoaded(true);
    }
  }
  fetchBossState();
  return () => { cancelled = true; };
}, []);
```

**Socket.IO event listeners (useEffect):**
```typescript
useEffect(() => {
  if (!socket) return;

  function onBossDamage(data: any) {
    setBoss((prev) =>
      prev && prev.id === data.bossfightId
        ? { ...prev, currentHp: data.currentHp, phase: data.phase }
        : prev
    );
    // Prepend to damage ticker
    setRecentDamage((prev) => [
      { id: crypto.randomUUID(), userName: data.userName, amount: 1, createdAt: new Date().toISOString() },
      ...prev.slice(0, 9), // Keep max 10
    ]);
    // Update leaderboard optimistically (will be corrected on next fetch)
    setLeaderboard((prev) => {
      const existing = prev.find((e) => e.userId === data.userId);
      if (existing) {
        const updated = prev.map((e) =>
          e.userId === data.userId ? { ...e, totalDamage: e.totalDamage + 1 } : e
        );
        return updated.sort((a, b) => b.totalDamage - a.totalDamage).slice(0, 3);
      }
      const newEntry = { userId: data.userId, userName: data.userName, totalDamage: 1 };
      return [...prev, newEntry].sort((a, b) => b.totalDamage - a.totalDamage).slice(0, 3);
    });
  }

  function onBossPhaseChange(data: any) {
    setBoss((prev) =>
      prev && prev.id === data.bossfightId
        ? { ...prev, phase: data.newPhase }
        : prev
    );
  }

  function onBossDefeated(data: any) {
    setVictory({
      bossName: boss?.name ?? "Boss",
      mvpUserName: data.mvpUserName ?? "Team",
      totalDamage: data.totalDamage ?? 0,
      runenEarned: data.runenEarned ?? 0,
    });
    setBoss(null);
  }

  function onBossSpawned(data: any) {
    setBoss({
      id: data.bossfightId,
      name: data.name,
      spawnHp: data.spawnHp,
      currentHp: data.spawnHp,
      phase: 1,
      spawnedAt: new Date().toISOString(),
    });
    setTeaser(null);
    setLeaderboard([]);
    setRecentDamage([]);
    setVictory(null);
  }

  function onBossHeal(data: any) {
    setBoss((prev) =>
      prev && prev.id === data.bossfightId
        ? { ...prev, currentHp: data.currentHp }
        : prev
    );
  }

  socket.on("boss:damage", onBossDamage);
  socket.on("boss:phase-change", onBossPhaseChange);
  socket.on("boss:defeated", onBossDefeated);
  socket.on("boss:spawned", onBossSpawned);
  socket.on("boss:heal", onBossHeal);

  return () => {
    socket.off("boss:damage", onBossDamage);
    socket.off("boss:phase-change", onBossPhaseChange);
    socket.off("boss:defeated", onBossDefeated);
    socket.off("boss:spawned", onBossSpawned);
    socket.off("boss:heal", onBossHeal);
  };
}, [socket, boss?.name, boss?.id]);
```
