Create `src/app/api/akten/[id]/feed/route.ts` with a GET handler:

**Route handler:**
```typescript
import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/lib/db";
import { requireAuth } from "@/lib/auth/require-auth";
import { buildAkteAccessFilter } from "@/lib/auth/akte-access";
import { AktenActivityTyp, Prisma } from "@prisma/client";
```

**Query params:**
- `cursor` (optional): AktenActivity ID for cursor pagination
- `take` (optional, default 20, max 100): number of entries per page
- `typ` (optional): comma-separated AktenActivityTyp values for filtering

**Implementation:**
1. Await `requireAuth()` for session
2. Await params to get `akteId` (Next.js 14 dynamic route params)
3. Verify akte access using `buildAkteAccessFilter` -- query the Akte with the access filter to confirm access, return 403 if not found
4. Parse query params (take clamped to `Math.min(parseInt(take ?? "20"), 100)`)
5. Build `where` filter:
   ```typescript
   const where: Prisma.AktenActivityWhereInput = { akteId };
   if (typ) {
     const types = typ.split(",").map(s => s.trim()).filter(Boolean) as AktenActivityTyp[];
     if (types.length === 1) {
       where.typ = types[0];
     } else if (types.length > 1) {
       where.typ = { in: types };
     }
   }
   ```
6. Query with cursor pagination:
   ```typescript
   const entries = await prisma.aktenActivity.findMany({
     where,
     orderBy: { createdAt: "desc" },
     take: take + 1, // fetch one extra to determine hasMore
     ...(cursor ? { cursor: { id: cursor }, skip: 1 } : {}),
     include: { user: { select: { id: true, name: true } } },
   });
   ```
7. Determine hasMore and slice:
   ```typescript
   const hasMore = entries.length > take;
   const items = hasMore ? entries.slice(0, take) : entries;
   ```
8. Return JSON:
   ```typescript
   return NextResponse.json({
     items,
     nextCursor: hasMore ? items[items.length - 1]?.id : null,
     hasMore,
   });
   ```

**Error handling:**
- 401 if not authenticated
- 403 if no akte access
- 500 with error log on unexpected errors

function relativeTime(dateStr: string): string {
  const diff = Date.now() - new Date(dateStr).getTime();
  const mins = Math.floor(diff / 60000);
  if (mins < 1) return "gerade eben";
  if (mins < 60) return `vor ${mins} Min.`;
  const hours = Math.floor(mins / 60);
  if (hours < 24) return `vor ${hours} Std.`;
  const days = Math.floor(hours / 24);
  if (days < 7) return `vor ${days} Tagen`;
  return new Date(dateStr).toLocaleDateString("de-DE", { day: "2-digit", month: "2-digit", year: "numeric" });
}
