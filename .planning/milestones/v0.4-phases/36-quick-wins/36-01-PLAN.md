```typescript
"use client";
import { useState } from "react";

interface AkteDetailClientProps {
  akte: AkteData; // Same AkteData type
  children?: React.ReactNode; // for NormenSection, etc.
}

export function AkteDetailClient({ akte }: AkteDetailClientProps) {
  const [activeTab, setActiveTab] = useState("feed");

  const tabMap: Record<string, string> = {
    "Beteiligte": "feed",
    "Dokumente": "dokumente",
    "Termine/Fristen": "kalender",
    "E-Mails": "email",   // no email tab exists, no-op
    "Zeiterfassung": "finanzen",
    "Chat": "nachrichten",
  };

  const handleKpiClick = (label: string) => {
    const tab = tabMap[label];
    if (!tab) return;
    // E-Mails has no tab in AkteDetailTabs -- ignore click
    if (label === "E-Mails") return;
    setActiveTab(tab);
    // For Zeiterfassung, scroll after tab switch
    if (label === "Zeiterfassung") {
      setTimeout(() => {
        document.getElementById("zeiterfassung-section")
          ?.scrollIntoView({ behavior: "smooth" });
      }, 150);
    }
  };

  return (
    <>
      <div className="grid grid-cols-2 md:grid-cols-6 gap-3">
        {/* Render StatMini cards with onClick + hover */}
      </div>
      <AkteDetailTabs
        akte={akte}
        activeTab={activeTab}
        onTabChange={setActiveTab}
      />
    </>
  );
}
```
