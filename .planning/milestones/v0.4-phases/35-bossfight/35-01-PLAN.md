export const BOSS_NAMES = [

  "Papierdrache",
  "Der Paragraphenwurm",
  "Terminmonster",

/** Lucide icon name per boss phase */
export const PHASE_ICONS = {
  1: "Bug",
  2: "Flame",
  3: "Skull",
  4: "Swords",
} as const;

/** Phase transition thresholds (checked from highest to lowest) */
export const PHASE_THRESHOLDS = [
  { phase: 4, hpFraction: 0.25 },
  { phase: 3, hpFraction: 0.50 },
  { phase: 2, hpFraction: 0.75 },
] as const;

/** Per-hit Runen multiplier per boss phase */
export const PHASE_RUNEN_MULTIPLIER: Record<number, number> = {
  1: 1.0,
  2: 1.5,
  3: 2.0,
  4: 3.0,
};

/** Base Runen earned per WV clear during an active bossfight */
export const BASE_HIT_RUNEN = 5;

/** Team-wide bonus Runen at each phase transition */
export const PHASE_TRANSITION_BONUS: Record<number, number> = {
  2: 20,
  3: 40,
  4: 80,
};

/** Victory rewards */
export const VICTORY_XP_BONUS = 500;
export const VICTORY_RUNEN_BONUS = 100;

/** Default boss config */
export const DEFAULT_BOSS_THRESHOLD = 30;
export const DEFAULT_BOSS_COOLDOWN_HOURS = 24;

/**
 * Calculate the boss phase based on current HP fraction.
 * Phase 4 at <=25%, Phase 3 at <=50%, Phase 2 at <=75%, Phase 1 otherwise.
 */
export function calculatePhase(currentHp: number, spawnHp: number): number {
  if (spawnHp <= 0) return 1;
  const fraction = currentHp / spawnHp;
  for (const threshold of PHASE_THRESHOLDS) {
    if (fraction <= threshold.hpFraction) return threshold.phase;
  }
  return 1;
}
```

/**
 * Award a boss victory trophy to a user's GameProfile.
 * Appends to the trophies JSON array.
 */
export async function awardTrophy(userId: string, trophy: BossTrophy): Promise<void> {
  const profile = await prisma.userGameProfile.findUnique({
    where: { userId },
    select: { trophies: true },
  });
  if (!profile) return;

  const existing = (profile.trophies as BossTrophy[] | null) ?? [];
  // Prevent duplicate trophy for the same bossfight
  if (existing.some((t) => t.bossfightId === trophy.bossfightId)) return;

  await prisma.userGameProfile.update({
    where: { userId },
    data: {
      trophies: [...existing, trophy],
    },
  });
}
```

**`getBacklogCount(kanzleiId: string): Promise<number>`**

**`getActiveBoss(kanzleiId: string): Promise<Bossfight | null>`**

**`checkAndSpawnBoss(kanzleiId: string): Promise<Bossfight | null>`**
- Get threshold from `getSettingTyped<number>("gamification.boss.threshold", DEFAULT_BOSS_THRESHOLD)`
- Get cooldown from `getSettingTyped<number>("gamification.boss.cooldownHours", DEFAULT_BOSS_COOLDOWN_HOURS)`
- Check if active boss already exists -- if so, return null
- Check last defeated boss: `prisma.bossfight.findFirst({ where: { kanzleiId, status: "DEFEATED" }, orderBy: { defeatedAt: "desc" } })`
- If last defeated exists and `differenceInHours(new Date(), lastDefeated.defeatedAt!) < cooldown`, return null (cooldown not elapsed)
- Count open WVs via `getBacklogCount(kanzleiId)`
- If count <= threshold, return null
- Spawn boss inside `$transaction`: double-check no active boss exists (prevent race), then create:
  - `name`: random pick from `BOSS_NAMES`
  - `spawnHp`: the open WV count
  - `currentHp`: same as spawnHp
  - `kanzleiId`
- Emit `boss:spawned` event to `kanzlei:{kanzleiId}` room with boss data
- Return the new boss

**`dealBossDamage(kanzleiId: string, userId: string, userName: string): Promise<DamageResult | null>`**
- Find active boss for kanzlei -- if none, return null
- Inside `$transaction`:
  - Check `boss.status === "ACTIVE"` (guard against post-defeat phantom damage)
  - Decrement HP: `tx.bossfight.update({ where: { id }, data: { currentHp: { decrement: 1 } } })`
  - Record damage: `tx.bossfightDamage.create({ ... })`
  - Calculate per-hit Runen: `Math.round(BASE_HIT_RUNEN * PHASE_RUNEN_MULTIPLIER[boss.phase])`
  - Set `runenEarned` on the BossfightDamage record
  - Award Runen to attacker via `awardRewards(userId, 0, hitRunen, 1)` (XP=0 for per-hit, streakMul=1 since boss rewards don't use streak)
  - Check phase transition: `calculatePhase(updatedBoss.currentHp, updatedBoss.spawnHp)`
  - If new phase > current phase:
    - Update boss phase
    - Check bitmask `phaseRewardsGiven` to see if this phase bonus was already awarded (bit = `1 << newPhase`)
    - If not yet awarded: get all participants from `tx.bossfightDamage.findMany({ where: { bossfightId }, select: { userId: true }, distinct: ["userId"] })`
    - Award `PHASE_TRANSITION_BONUS[newPhase]` Runen to each participant via loop of `awardRewards`
    - Update bitmask: `phaseRewardsGiven: { set: boss.phaseRewardsGiven | (1 << newPhase) }`
    - Emit `boss:phase-change` event
  - Check defeat: if `updatedBoss.currentHp <= 0`:
    - Update boss: `status: "DEFEATED", defeatedAt: new Date(), currentHp: 0`
    - Call `awardVictoryRewards(tx, bossfightId)` (see below)
    - Emit `boss:defeated` event with MVP (top damage dealer), total damage, Runen earned
  - Else emit `boss:damage` event with { bossfightId, userId, userName, currentHp, spawnHp, phase }
- Return result object

**`healBoss(kanzleiId: string): Promise<void>`**
- Find active boss -- if none, return
- Inside `$transaction`:
  - Check `boss.currentHp < boss.spawnHp` (cap at spawnHp)
  - Increment: `tx.bossfight.update({ where: { id }, data: { currentHp: { increment: 1 } } })`
  - Only increment if currentHp < spawnHp (use a raw where clause: `AND: [{ currentHp: { lt: boss.spawnHp } }]`)
- Emit `boss:heal` event to kanzlei room with updated HP

**Private helper `awardVictoryRewards(tx, bossfightId)`:**
- Get all distinct participant userIds from BossfightDamage
- For each participant:
  - `awardRewards(userId, VICTORY_XP_BONUS, VICTORY_RUNEN_BONUS, 1)` (no streak multiplier on boss rewards)
  - `awardTrophy(userId, { type: "BOSS_VICTORY", bossName: boss.name, date: new Date().toISOString(), bossfightId })`
- Get MVP (user with most total damage): `tx.bossfightDamage.groupBy({ by: ["userId"], where: { bossfightId }, _sum: { amount: true }, orderBy: { _sum: { amount: "desc" } }, take: 1 })`
